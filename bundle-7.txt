"use client";

import { useEffect, useState } from "react";
import { useRefresh } from "@/components/providers/RefreshProvider";

export function useFetch<T>(url: string) {
  const { refreshTick } = useRefresh();
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    let cancelled = false;
    setLoading(true);
    fetch(url)
      .then((r) => r.json())
      .then((d) => {
        if (!cancelled) {
          setData(d);
          setError(null);
        }
      })
      .catch((e) => {
        if (!cancelled) setError(String(e));
      })
      .finally(() => {
        if (!cancelled) setLoading(false);
      });
    return () => {
      cancelled = true;
    };
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [url, refreshTick]);

  return { data, loading, error };
}

"use client";

import { useFetch } from "@/lib/useFetch";
import { KpiCard } from "@/components/ui/KpiCard";
import { Gauge } from "@/components/ui/Gauge";
import { AlertCard } from "@/components/alerts/AlertCard";
import { ChartCard } from "@/components/charts/ChartCard";
import { formatCurrency, formatNumber } from "@/lib/format";
import {
  Package,
  DollarSign,
  TrendingUp,
  Activity,
  Clock,
  AlertTriangle,
  ShoppingCart,
  CheckCircle2,
  Truck,
  Wallet,
  LayoutGrid,
} from "lucide-react";

export default function DashboardPage() {
  const { data: inv } = useFetch<{ items: any[] }>("/api/inventory");
  const { data: analytics } = useFetch<any>("/api/analytics");
  const { data: alertsData } = useFetch<{ alerts: any[] }>("/api/alerts");

  const kpis = analytics?.kpis;
  const healthScore = analytics?.healthScore ?? 0;
  const topAlerts = (alertsData?.alerts || []).slice(0, 6);

  const categoryOption = analytics
    ? {
        tooltip: { trigger: "item" },
        legend: { bottom: 0, textStyle: { fontSize: 10 } },
        series: [
          {
            type: "pie",
            radius: ["40%", "70%"],
            data: Object.entries(analytics.byCategory).map(([name, v]: any) => ({
              name,
              value: +v.value.toFixed(0),
            })),
          },
        ],
      }
    : {};

  const statusOrder = ["Order Now", "In Stock", "On Order", "Over Stock", "Critical"];
  const statusOption = inv
    ? {
        tooltip: { trigger: "axis" },
        xAxis: { type: "category", data: statusOrder },
        yAxis: { type: "value" },
        series: [
          {
            type: "bar",
            data: statusOrder.map((s) => inv.items.filter((i: any) => i.orderStatus === s).length),
            itemStyle: { color: "#2E6BE6", borderRadius: [6, 6, 0, 0] },
          },
        ],
      }
    : {};

  return (
    <div className="space-y-6">
      <div>
        <h1 className="text-xl font-bold text-slate-800 dark:text-white">Executive Dashboard</h1>
        <p className="text-sm text-erp-muted">Real-time overview of inventory health, cost, and risk.</p>
      </div>

      <div className="grid grid-cols-2 md:grid-cols-3 xl:grid-cols-6 gap-4">
        <KpiCard title="Total Items" value={formatNumber(kpis?.totalItems ?? 0)} icon={Package} tone="navy" />
        <KpiCard title="Inventory Value" value={formatCurrency(kpis?.inventoryValue ?? 0)} icon={DollarSign} tone="success" />
        <KpiCard title="Monthly Stock Cost" value={formatCurrency(kpis?.totalMonthlyStockCost ?? 0)} icon={Wallet} tone="accent" />
        <KpiCard title="Monthly Demand" value={formatNumber(kpis?.totalMonthlyDemand ?? 0)} icon={TrendingUp} tone="accent" />
        <KpiCard title="Avg Coverage" value={`${formatNumber(kpis?.avgCoverage ?? 0)} d`} icon={Activity} tone="muted" />
        <KpiCard title="Avg Lead Time" value={`${formatNumber(kpis?.avgLeadTime ?? 0)} d`} icon={Clock} tone="muted" />
        <KpiCard title="Critical Items" value={formatNumber(kpis?.criticalItems ?? 0)} icon={AlertTriangle} tone="critical" />
        <KpiCard title="Items To Order" value={formatNumber(kpis?.itemsToOrder ?? 0)} icon={ShoppingCart} tone="warning" />
        <KpiCard title="Healthy Items" value={formatNumber(kpis?.healthyItems ?? 0)} icon={CheckCircle2} tone="success" />
        <KpiCard title="Suppliers" value={formatNumber(kpis?.suppliers ?? 0)} icon={Truck} tone="navy" />
        <KpiCard title="Payment Required" value={formatCurrency(kpis?.paymentRequired ?? 0)} icon={Wallet} tone="warning" />
        <KpiCard title="Categories" value={formatNumber(kpis?.categories ?? 0)} icon={LayoutGrid} tone="navy" />
      </div>

      <div className="grid grid-cols-1 lg:grid-cols-3 gap-4">
        <div className="bg-erp-card dark:bg-slate-800 rounded-xl2 border border-erp-border dark:border-slate-700 shadow-card p-4 flex flex-col items-center justify-center">
          <h3 className="text-sm font-semibold text-slate-700 dark:text-slate-200 mb-2 self-start">
            Inventory Health Score
          </h3>
          <Gauge value={healthScore} label="out of 100" />
        </div>
        <div className="lg:col-span-2">
          <ChartCard title="Inventory Value by Category" option={categoryOption} />
        </div>
      </div>

      <div className="grid grid-cols-1 lg:grid-cols-3 gap-4">
        <div className="lg:col-span-2">
          <ChartCard title="Items by Order Status" option={statusOption} />
        </div>
        <div className="bg-erp-card dark:bg-slate-800 rounded-xl2 border border-erp-border dark:border-slate-700 shadow-card p-4">
          <h3 className="text-sm font-semibold text-slate-700 dark:text-slate-200 mb-3">Recent Alerts</h3>
          <div className="space-y-2 max-h-72 overflow-y-auto pr-1">
            {topAlerts.length === 0 && <p className="text-xs text-erp-muted">No alerts. Everything looks healthy.</p>}
            {topAlerts.map((a: any) => (
              <AlertCard key={a.id} alert={a} />
            ))}
          </div>
        </div>
      </div>
    </div>
  );
}

"use client";

import { useSearchParams } from "next/navigation";
import { useFetch } from "@/lib/useFetch";
import { InventoryGrid } from "@/components/inventory/InventoryGrid";

export function InventoryPageClient() {
  const params = useSearchParams();
  const q = params.get("q") || "";
  const { data, loading } = useFetch<{ items: any[] }>("/api/inventory");

  return (
    <div className="space-y-4">
      <div>
        <h1 className="text-xl font-bold text-slate-800 dark:text-white">Inventory</h1>
        <p className="text-sm text-erp-muted">Full item master with sorting, filtering, column tools, and export.</p>
      </div>
      {loading && !data ? (
        <p className="text-sm text-erp-muted">Loading inventory...</p>
      ) : (
        <InventoryGrid items={data?.items || []} initialSearch={q} />
      )}
    </div>
  );
}
